(1) Add Loader For Style File
--------------------------------
==> For Includes The Style Files In Latest Bundle After Build
==> Install Both Loaders [npm i css-loader style-loader]
==> In 'webpack.config.js' Add The Loaders


    style/style.css
    -----------------

    ...


    index.js
    -----------


    import png from "./assets/png.png"
    import './style/style.css' /////////// Note Import Here //////////////////

    const btn = document.getElementById('my-btn')
    const logo = document.getElementById('logo')


    btn.addEventListener('click', () => {
    
        logo.src = png
        
    })

    webpack.config.js
    -------------------

    module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /.(css)$/,
                    use: ["style-loader" , "css-loader"]
                }
            ]
        }


(2) Add Sass Loader
---------------------
==> npm i --save-dev sass-loader sass

    style/sassy.scss
    ------------------

    ....



    webpack.config.js
    -------------------

    module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /\.css$/,
                    use: ["style-loader" , "css-loader"]
                },
                {
                    test: /.scss$/i,
                    use: [
                        "style-loader",
                        "css-loader",
                        "sass-loader"
                    ],
                },
            ]
        }


    index.js
    ----------

    import png from "./assets/png.png"
    import './style/style.css'
    import './style/sassy.scss'

    const btn = document.getElementById('my-btn')
    const logo = document.getElementById('logo')


    btn.addEventListener('click', () => {
    
        logo.src = png
        
    })


==> Run [npm run build] And see if the 'sassy.scss' outputed in 'dist/bundle.js'
