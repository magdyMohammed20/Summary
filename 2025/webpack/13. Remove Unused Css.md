(1) Remove Unused Css
----------------------
==> We Need To Remove The Css Code That Not Used In Any Html
==> Install : [npm i purgecss-webpack-plugin --save-dev]


    webpack.prod.config.js
    -------------------------

    const path = require("path");

    // 1- Import glob
    const glob = require('glob');
    const HtmlPlugin = require('html-webpack-plugin');
    const CopyPlugin = require('copy-webpack-plugin');
    const MiniCssExtractPlugin = require('mini-css-extract-plugin');
    const CssMinimizerPlugin = require('css-minimizer-webpack-plugin');

    // 2- Import The Plugin
    const { PurgeCSSPlugin } = require('purgecss-webpack-plugin');

    // 3- Prepare PATHS
    const PATHS = {
        root: path.join(__dirname) // If Files Inside 'src' Folder Will Be Like path.join(__dirname , 'src')
    };

    module.exports = {
        mode: "production",
        entry: {
            index: {
                import: "./index.js",
                filename: "main-entry.js"
            },
            explore: "./explore.js"
        },
        optimization: {
            minimizer: [`...`, new CssMinimizerPlugin()]
        },
        output: {
            filename: "[name].bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]",
            clean: true
        },

        plugins: [
            // 4- Usage Of PurgeCSSPlugin
            new PurgeCSSPlugin({
                paths: glob.sync(`${PATHS.root}/**/*`, { nodir: true }),
            }),
            new MiniCssExtractPlugin(),
            new HtmlPlugin({
                template: "./index.html",
                chunks: ["index"],
                filename: "index.html"
            }),
            new HtmlPlugin({
                template: "./explore.html",
                chunks: ["explore"],
                filename: "explore.html"
            }),
            new CopyPlugin({
                patterns: [
                    {
                        from: path.resolve(__dirname, 'assets'),
                        to: path.resolve(__dirname, "dist", "assets/"),
                    }
                ]
            })
        ],

        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /\.css$/,
                    use: [MiniCssExtractPlugin.loader, "css-loader"]
                },
                {
                    test: /.scss$/i,
                    use: [
                        MiniCssExtractPlugin.loader,
                        "css-loader",
                        "sass-loader"
                    ],
                },
                {
                    test: /\.(ttf)$/,
                    type: "asset/resource"
                }
            ]
        }
    };


(2) Specify Files To Remove Unused Css From It Only
-----------------------------------------------------


    new PurgeCSSPlugin({
        paths: glob.sync(`${PATHS.root}/**/*`, { nodir: true }),
        only: ['explore'] // Will Remove Unused Css Only From The explore.css Only 
    })


(3) If Need To Prevent Remove Of Specific Class
--------------------------------------------------

    new PurgeCSSPlugin({
        paths: glob.sync(`${PATHS.root}/**/*`, { nodir: true }),
        only: ['explore'],
        safelist: ['magdy'] // When Remove Unused Css Don't Remove The Unused Class That Called 'magdy'
    })
