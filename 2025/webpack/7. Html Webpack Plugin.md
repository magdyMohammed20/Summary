(1) Html Webpack Plugin
--------------------------
==> Install The Plugin [npm i html-webpack-plugin --save-dev]
==> Start Include The Plugin in 'webpack.config.js'

    webpack.config.js
    --------------------

    const path = require("path")

    // 1- Import The Html Plugin
    const HtmlPlugin = require('html-webpack-plugin')

    module.exports = {
        entry: {
            index: {
                import: "./index.js",
                filename: "main-entry.js"
            },
            explore: "./explore.js"
        },
        output: {
            filename: "[name].bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]",
            clean: true
        },

        // 2- Include The Plugin Here
        // template: For Render The Html Content In Generated Html File In 'dist' Folder 
        // chunks: ["index"] → only the index entry (main-entry.js) is added to your index.html
        plugins: [
            new HtmlPlugin({
                template: "./index.html",
                chunks: ["index"]

            })
        ],
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /\.css$/,
                    use: ["style-loader" , "css-loader"]
                },
                {
                    test: /.scss$/i,
                    use: [
                        "style-loader",
                        "css-loader",
                        "sass-loader"
                    ],
                },
                {
                    test: /\.(ttf)$/,
                    type: "asset/resource"
                }
            ]
        }
    }
    


    index.html (Don't Need To Include The Script Here As It Will Be Included In The Generated Html File In 'dist' After Build)
    ----------------
    
    <!DOCTYPE html>
    <html lang="en">
        <head>
            <meta charset="UTF-8">
            <meta name="viewport" content="width=device-width, initial-scale=1.0">
            <title>Document</title>
        </head>
        <body>
            
            <button id="my-btn">Click</button>
            <img src="" id="logo"/>

            <h1>Heading Title</h1>
        </body>
    </html>


(2) Multi HtmlPlugin Usage 
--------------------------------

    webpack.config.js
    --------------------
    
    const path = require("path")

    // 1- Import The Html Plugin
    const HtmlPlugin = require('html-webpack-plugin')

    module.exports = {
        entry: {
            index: {
                import: "./index.js",
                filename: "main-entry.js"
            },
            explore: "./explore.js"
        },
        output: {
            filename: "[name].bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]",
            clean: true
        },

        // 2- Include The Plugin Here
        // template: For Render The Html Content In Generated Html File In 'dist' Folder 
        // chunks: ["index"] → only the index entry (main-entry.js) is added to your index.html
        plugins: [
            new HtmlPlugin({
                template: "./index.html",
                chunks: ["index"],
                filename: "index.html"   // ✅ output name

            }),
            new HtmlPlugin({
                template: "./explore.html",
                chunks: ["explore"],
                filename: "explore.html" // ✅ different output name
            })
        ],
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /\.css$/,
                    use: ["style-loader" , "css-loader"]
                },
                {
                    test: /.scss$/i,
                    use: [
                        "style-loader",
                        "css-loader",
                        "sass-loader"
                    ],
                },
                {
                    test: /\.(ttf)$/,
                    type: "asset/resource"
                }
            ]
        }
    }
