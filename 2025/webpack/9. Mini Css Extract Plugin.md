(1) Mini Css Extract Plugin
-----------------------------
==> This plugin extracts CSS into separate files. It creates a CSS file for each JS file 
    that contains CSS. It supports On-Demand-Loading of CSS and SourceMaps.
==> npm i mini-css-extract-plugin --save-dev



    webpack.config.JS
    -------------------

    const path = require("path")

    const HtmlPlugin = require('html-webpack-plugin')

    const CopyPlugin = require('copy-webpack-plugin')

    // 1- Import The Lib
    const MiniCssExtractPlugin = require('mini-css-extract-plugin')

    module.exports = {
        entry: {
            index: {
                import: "./index.js",
                filename: "main-entry.js"
            },
            explore: "./explore.js"
        },
        output: {
            filename: "[name].bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]",
            clean: true
        },


        plugins: [
            // 2- Import In Plugins Here
            new MiniCssExtractPlugin(),
            new HtmlPlugin({
                template: "./index.html",
                chunks: ["index"],
                filename: "index.html" 

            }),
            new HtmlPlugin({
                template: "./explore.html",
                chunks: ["explore"],
                filename: "explore.html" 
            }),

            new CopyPlugin({
                patterns: [
                    {
                        // 4- Remove assets/* and set assets only 
                        from: path.resolve(__dirname, 'assets'),

                        to: path.resolve(__dirname, "dist", "assets/"),
                        
                    
                    }
                ]
            })
        ],
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /\.css$/,
                    use: [MiniCssExtractPlugin.loader , "css-loader"] // 3- Replace style-loader with the instance
                },
                {
                    test: /.scss$/i,
                    use: [
                        MiniCssExtractPlugin.loader, // 3- Replace style-loader with the instance
                        "css-loader",
                        "sass-loader"
                    ],
                },
                {
                    test: /\.(ttf)$/,
                    type: "asset/resource"
                }
            ]
        }
    }
    
    index.js
    -----------
    
    import './style/style.css' // Will Generated As 'index.css'
    import './style/sassy.scss'
    

    explore.js
    ------------
    

    import png from "./assets/png.png"
    import './style/style.css' // Will Generated As 'explore.css'
