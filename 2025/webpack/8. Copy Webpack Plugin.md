1- Copy Webpack Plugin
--------------------------
==> If Uses Static Images In Html It Will Not Included At Final Bundle As Assets
    Or You Have File Contains Images And Not Included In Final Bundle And Need To Move It
    So Will Use This Plugin To Copy It To 'dist' Folder


2- Usage
-----------
==> Install : [npm i copy-webpack-plugin --save-dev]
==> Handle Config In 'webpack.config.js'


    webpack.config.js
    --------------------


    const path = require("path")

    const HtmlPlugin = require('html-webpack-plugin')

    // 1- Import The Plugin
    const CopyPlugin = require('copy-webpack-plugin')


    module.exports = {
        entry: {
            index: {
                import: "./index.js",
                filename: "main-entry.js"
            },
            explore: "./explore.js"
        },
        output: {
            filename: "[name].bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]",
            clean: true
        },


        plugins: [
            new HtmlPlugin({
                template: "./index.html",
                chunks: ["index"],
                filename: "index.html" 

            }),
            new HtmlPlugin({
                template: "./explore.html",
                chunks: ["explore"],
                filename: "explore.html" 
            }),

            // 2- Use Plugin Like This
            new CopyPlugin({
                patterns: [
                    {
                        // This Is Current Position Of Needed Moved Folder
                        from: path.resolve(__dirname, 'assets/*'),

                        // This Is The Final Destination Of It (dist/assets)
                        to: path.resolve(__dirname, "dist", "assets/"),
                    }
                ]
            })
        ],
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                },
                {
                    test: /\.css$/,
                    use: ["style-loader" , "css-loader"]
                },
                {
                    test: /.scss$/i,
                    use: [
                        "style-loader",
                        "css-loader",
                        "sass-loader"
                    ],
                },
                {
                    test: /\.(ttf)$/,
                    type: "asset/resource"
                }
            ]
        }
    }
    
3- Other Copy Plugin Options
-------------------------------


    new CopyPlugin({
        patterns: [
            {
                from: path.resolve(__dirname, 'assets/*'),
                to: path.resolve(__dirname, "dist", "assets/"),
                
                // Defines the base directory used for two purposes
                //  1- It is prepended to the from path
                //  2- It is removed from the beginning of the result path(s).
                context, 


                // Allows you to modify the contents of a file before it is written to the output directory.
                // Like Optimize The Images
                transform,

                // Overwrites files that already exist in compilation.assets (typically added by other plugins or loaders).
                force,

                // Doesn't generate an error if file(s) are missing.
                noErrorOnMissing,
            }
        ]
    })
