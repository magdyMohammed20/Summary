(1) Loaders
---------------
==> Responsible For Transpiling , Manage Images , Fonts And Css Files

(2) Create Loader For Images To Include It In Final Bundle
------------------------------------------------------------


    index.html
    -------------

        <button id="my-btn">Click</button>
        <img src="" id="logo"/>
        <script src="./dist/bundle.js"></script>


    webpack.config.js (Here We Add Loaders For Images With Images Extensions )
    -------------------

    const path = require("path")

    module.exports = {
        entry: "./index.js",
        output: {
            filename: "bundle.js",
            path: path.resolve(__dirname , "dist")
        },
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                }
            ]
        }
    }


(3) For The Previous Configuration The Images Setted Directed Inside 'dist' Folder
    So If Need To Handle The Output Files And Set The Images In Path Like '/assets/png.png' Make The Following
---------------------------------------------------------------------------------------
==> Add 'assetModuleFilename'


    webpack.config.js (Here We Add Loaders For Images With Images Extensions )
    -------------------


    const path = require("path")

    module.exports = {
        entry: "./index.js",
        output: {
            filename: "bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]"
        },
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                }
            ]
        }
    }


(4) For Override All 'dist' Folder Add 'Clean' Property
------------------------------------------------------------

    webpack.config.js
    -------------------

    const path = require("path")

    module.exports = {
        entry: "./index.js",
        output: {
            filename: "bundle.js",
            path: path.resolve(__dirname, "dist"),
            assetModuleFilename: "asset/[hash][ext]",
            clean: true
        },
        module: {
            rules: [
                {
                    test: /.(png|jpe?g)$/,
                    type: "asset/resource"
                }
            ]
        }
    }
