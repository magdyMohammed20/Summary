            let   one   = 1,
                  two   = 2,
                  three = 3
            let arr = [1,2,3];

            let   one   = 1,
                  two   = 2,
                  three = 3
            const arr = [1,2,3];
            [one,two,three] = arr; // Remove const Keyword But If Variables Above Is Const This Statement Will Be Error

            console.log(one)// 1
