import React,{useState,useEffect} from 'react';

const Comp3 = () => {

    const [counter , setCounter] = useState(0)

    useEffect(()=>{
        const interval = setInterval(()=>setCounter(counter + 1),1000)
        return ()=>{
            clearInterval(interval) // Should Clear Interval Here
        }
    })

    return (
        <div>
            {counter}
        </div>
    );
};

export default Comp3;
