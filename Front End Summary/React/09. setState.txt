(1) Use setState() Function For Change Specific Value In Component State
---------------------------------------------------------------------------

	class Child extends Component{
	
		state={
			name: 'Magdy'
		}

  		print = () =>{
			// Change name Value
    			this.setState({name: 'Mohammed'})
  		}	

		render(){
			return(
				<div>
				  <button onClick={this.print}>
				    Click
				  </button>
				  {this.state.name}
				</div>
			)
		}

	}
