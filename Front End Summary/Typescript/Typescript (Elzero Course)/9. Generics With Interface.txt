(1) Generics With Interface
------------------------------

    interface Book {
        title: string;
        pages: number;
        hasColorCover: boolean;
    }

    class KidsBook<T> {
        public data: T[] = []
        add(item: T) {
            this.data.push(item)
        }
    }

    const bo = new KidsBook<Book>()
    bo.add({ title: 'Title', pages: 20, hasColorCover: true })
