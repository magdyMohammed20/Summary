(1) Interface ReOpen
--------------------------
==> You Can Add Members To Interface By ReDeclare It

Ex
---

    interface Settings {
        font: string;
        theme: string;
    }

    interface Settings {
        sidebar: boolean
    }

    const userSettings: Settings = {
        font: 'tahoma',
        theme: 'dark',
        sidebar: true
    }

    console.log(userSettings) // {font: 'tahoma', theme: 'dark', sidebar: true}
