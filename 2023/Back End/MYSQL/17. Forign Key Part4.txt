(2) One To One
---------------
==> Like Visa Card That Each Client Has Unique Card Number

EX
---

    CREATE TABLE cards(
        card_id INT NOT NULL,
        card_num VARCHAR(255) UNIQUE,
        client_id INT NOT NULL,
        PRIMARY KEY(card_id),
        FOREIGN KEY(client_id) REFERENCES clients(id)

    ) ENGINE=INNODB;
